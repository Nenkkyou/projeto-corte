# Dev Panel + Client Auth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Adicionar auth de clientes (cadastro/login antes do formulário), rastreamento de eventos e painel DEV com monitoramento completo do sistema.

**Architecture:** Tudo persiste em localStorage com chaves prefixadas `pc_`. O `script.js` centraliza auth e tracking. `cadastro.html` é a porta de entrada dos clientes. `dev.html` é o painel isolado do desenvolvedor protegido por senha mestre.

**Tech Stack:** HTML/CSS/JS puro, Web Crypto API (SHA-256), localStorage, Vite (dev server)

```javascript
// --- Tracking ---
function genId() {
  return Date.now().toString(36) + Math.random().toString(36).slice(2, 7);
}

function trackEvent(type, detail = '') {
  const events = JSON.parse(localStorage.getItem('pc_events') || '[]');
  events.push({ id: genId(), type, detail, timestamp: new Date().toISOString() });
  if (events.length > 500) events.splice(0, events.length - 500);
  localStorage.setItem('pc_events', JSON.stringify(events));
}

async function hashPassword(password) {
  const msgBuffer = new TextEncoder().encode(password);
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
  return Array.from(new Uint8Array(hashBuffer)).map(b => b.toString(16).padStart(2, '0')).join('');
}
// --- End Tracking ---
```

## Task 2: Auth de clientes — funções no script.js

**Files:**
- Modify: `script.js`

- [ ] **Step 1: Adicionar constante de chave de sessão cliente e funções de auth cliente**

Logo após `const CREDENTIALS = ...`, adicionar:

```javascript
const CLIENT_SESSION_KEY = 'pc_client_session';

async function registerClient(name, email, password) {
  const users = JSON.parse(localStorage.getItem('pc_users') || '[]');
  if (users.find(u => u.email === email)) return { ok: false, error: 'Email já cadastrado.' };
  const passwordHash = await hashPassword(password);
  const user = { name, email, passwordHash, createdAt: new Date().toISOString() };
  users.push(user);
  localStorage.setItem('pc_users', JSON.stringify(users));
  localStorage.setItem(CLIENT_SESSION_KEY, JSON.stringify({ name, email, loginAt: new Date().toISOString() }));
  trackEvent('client_register', email);
  return { ok: true };
}

async function loginClient(email, password) {
  const users = JSON.parse(localStorage.getItem('pc_users') || '[]');
  const user = users.find(u => u.email === email);
  if (!user) { trackEvent('client_login_fail', email); return { ok: false, error: 'Email não encontrado.' }; }
  const hash = await hashPassword(password);
  if (hash !== user.passwordHash) { trackEvent('client_login_fail', email); return { ok: false, error: 'Senha incorreta.' }; }
  localStorage.setItem(CLIENT_SESSION_KEY, JSON.stringify({ name: user.name, email, loginAt: new Date().toISOString() }));
  trackEvent('client_login', email);
  return { ok: true };
}

async function handleCadastroSubmit(e) {
  e.preventDefault();
  const form = e.target;
  const tab = form.dataset.tab;
  const errorEl = document.getElementById('cadastroError');
  errorEl.style.display = 'none';

  if (tab === 'register') {
    const name = form.querySelector('#regName').value.trim();
    const email = form.querySelector('#regEmail').value.trim();
    const password = form.querySelector('#regPassword').value;
    if (password.length < 6) { errorEl.textContent = 'Senha deve ter no mínimo 6 caracteres.'; errorEl.style.display = 'flex'; return; }
    const result = await registerClient(name, email, password);
    if (!result.ok) { errorEl.textContent = result.error; errorEl.style.display = 'flex'; return; }
  } else {
    const email = form.querySelector('#loginEmail').value.trim();
    const password = form.querySelector('#loginPassword').value;
    const result = await loginClient(email, password);
    if (!result.ok) { errorEl.textContent = result.error; errorEl.style.display = 'flex'; return; }
  }
  window.location.href = 'proposta.html';
}
```

```html
    // DEV Panel logic
    function devLogout() {
      localStorage.removeItem('pc_dev_session');
      window.location.href = 'dev-login.html';
    }

    function fmtDate(iso) {
      if (!iso) return '—';
      const d = new Date(iso);
      return d.toLocaleDateString('pt-BR') + ' ' + d.toLocaleTimeString('pt-BR');
    }

    // Navigation
    document.querySelectorAll('.dev-nav-item').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelectorAll('.dev-nav-item').forEach(b => b.classList.remove('active'));
        document.querySelectorAll('.dev-section').forEach(s => s.classList.remove('active'));
        btn.classList.add('active');
        document.getElementById('section-' + btn.dataset.section).classList.add('active');
        renderSection(btn.dataset.section);
      });
    });

    function renderSection(name) {
      if (name === 'overview') renderOverview();
      else if (name === 'users') renderUsers();
      else if (name === 'submissions') renderSubmissions();
      else if (name === 'events') renderEvents();
      else if (name === 'storage') renderStorage();
    }

    // Overview
    function renderOverview() {
      const users = JSON.parse(localStorage.getItem('pc_users') || '[]');
      const submissions = JSON.parse(localStorage.getItem('pc_submissions') || '[]');
      const events = JSON.parse(localStorage.getItem('pc_events') || '[]');
      const lastEvent = events[events.length - 1];

      const stats = [
        { label: 'Clientes Cadastrados', value: users.length, sub: 'pc_users' },
        { label: 'Formulários Enviados', value: submissions.length, sub: 'pc_submissions' },
        { label: 'Eventos no Log', value: events.length, sub: 'máx 500' },
        { label: 'Último Evento', value: lastEvent ? lastEvent.type : '—', sub: lastEvent ? fmtDate(lastEvent.timestamp) : '' },
      ];
      document.getElementById('overviewStats').innerHTML = stats.map(s => `
        <div class="dev-stat-card">
          <div class="dev-stat-label">${s.label}</div>
          <div class="dev-stat-value">${s.value}</div>
          <div class="dev-stat-sub">${s.sub}</div>
        </div>`).join('');

      const recent = [...events].reverse().slice(0, 10);
      document.querySelector('#overviewRecentEvents tbody').innerHTML = recent.length
        ? recent.map(e => `<tr><td>${fmtDate(e.timestamp)}</td><td><span class="event-badge badge-${e.type}">${e.type}</span></td><td>${e.detail}</td></tr>`).join('')
        : '<tr><td colspan="3" class="dev-empty">Nenhum evento registrado</td></tr>';
    }

    // Users
    function renderUsers() {
      const users = JSON.parse(localStorage.getItem('pc_users') || '[]');
      document.querySelector('#usersTable tbody').innerHTML = users.length
        ? [...users].reverse().map(u => `<tr><td>${u.name}</td><td>${u.email}</td><td>${fmtDate(u.createdAt)}</td></tr>`).join('')
        : '<tr><td colspan="3" class="dev-empty">Nenhum usuário cadastrado</td></tr>';
    }

    // Submissions
    function renderSubmissions() {
      const submissions = JSON.parse(localStorage.getItem('pc_submissions') || '[]');
      const tbody = document.querySelector('#submissionsTable tbody');
      if (!submissions.length) { tbody.innerHTML = '<tr><td colspan="5" class="dev-empty">Nenhum formulário enviado</td></tr>'; return; }
      tbody.innerHTML = [...submissions].reverse().map(s => `
        <tr class="expandable-row" onclick="toggleDetail('${s.id}')">
          <td>${fmtDate(s.submittedAt)}</td>
          <td>${s.clientName}<br><span style="color:#6b7280;font-size:.75rem">${s.clientEmail}</span></td>
          <td>${s.data.projectName}</td>
          <td>${s.data.budget}</td>
          <td><i class="fas fa-chevron-down" style="color:#6b7280"></i></td>
        </tr>
        <tr class="detail-row" id="detail-${s.id}">
          <td colspan="5">${JSON.stringify(s.data, null, 2)}</td>
        </tr>`).join('');
    }

    function toggleDetail(id) {
      const row = document.getElementById('detail-' + id);
      row.classList.toggle('open');
    }

    // Events
    function renderEvents() {
      const filter = document.getElementById('eventFilter').value;
      const events = JSON.parse(localStorage.getItem('pc_events') || '[]');
      const filtered = filter ? events.filter(e => e.type === filter) : events;
      document.querySelector('#eventsTable tbody').innerHTML = filtered.length
        ? [...filtered].reverse().map(e => `<tr><td style="font-family:var(--font-mono);font-size:.8rem">${fmtDate(e.timestamp)}</td><td><span class="event-badge badge-${e.type}">${e.type}</span></td><td>${e.detail}</td></tr>`).join('')
        : '<tr><td colspan="3" class="dev-empty">Nenhum evento</td></tr>';
    }

    function clearEvents() {
      if (!confirm('Limpar todo o event log?')) return;
      localStorage.removeItem('pc_events');
      renderEvents();
      renderOverview();
    }

    // HealthCheck
    const HEALTH_PAGES = ['index.html', 'proposta.html', 'cadastro.html', 'login.html', 'dashboard.html', 'clientes.html', 'projetos.html', 'agenda.html', 'dev.html'];
    const HEALTH_ASSETS = ['assets/logo.png', 'style.css'];

    async function runHealthChecks() {
      const grid = document.getElementById('healthGrid');
      const checks = [];

      // Pages
      for (const page of HEALTH_PAGES) {
        checks.push({ label: `Página: ${page}`, key: page, type: 'page' });
      }
      // Assets
      for (const asset of HEALTH_ASSETS) {
        checks.push({ label: `Asset: ${asset}`, key: asset, type: 'asset' });
      }
      // localStorage
      checks.push({ label: 'localStorage funcional', key: 'ls', type: 'ls' });
      // JS errors últimas 24h
      checks.push({ label: 'Erros JS (últimas 24h)', key: 'jserr', type: 'jserr' });

      grid.innerHTML = checks.map(c => `
        <div class="health-item" id="hc-${c.key.replace(/[^a-z0-9]/gi,'-')}">
          <i class="health-icon pending fas fa-circle-notch fa-spin"></i>
          <div><div class="health-label">${c.label}</div><div class="health-msg">verificando...</div></div>
        </div>`).join('');

      for (const c of checks) {
        const id = 'hc-' + c.key.replace(/[^a-z0-9]/gi, '-');
        let ok = false, msg = '';
        try {
          if (c.type === 'page' || c.type === 'asset') {
            const res = await fetch(c.key, { cache: 'no-store' });
            ok = res.ok;
            msg = ok ? `HTTP ${res.status}` : `HTTP ${res.status} — falha`;
          } else if (c.type === 'ls') {
            localStorage.setItem('__hc_test__', '1');
            ok = localStorage.getItem('__hc_test__') === '1';
            localStorage.removeItem('__hc_test__');
            msg = ok ? 'leitura/escrita OK' : 'falha ao ler/escrever';
          } else if (c.type === 'jserr') {
            const events = JSON.parse(localStorage.getItem('pc_events') || '[]');
            const cutoff = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString();
            const count = events.filter(e => e.type === 'js_error' && e.timestamp > cutoff).length;
            ok = count === 0;
            msg = ok ? 'nenhum erro nas últimas 24h' : `${count} erro(s) detectado(s)`;
          }
        } catch (err) {
          ok = false; msg = err.message;
        }
        const el = document.getElementById(id);
        el.querySelector('.health-icon').className = `health-icon ${ok ? 'ok' : 'fail'} fas fa-${ok ? 'circle-check' : 'circle-xmark'}`;
        el.querySelector('.health-msg').textContent = msg;
      }
    }

    // LocalStorage Viewer
    function renderStorage() {
      const keys = Object.keys(localStorage).sort();
      const el = document.getElementById('storageList');
      if (!keys.length) { el.innerHTML = '<div class="dev-empty">localStorage vazio</div>'; return; }
      el.innerHTML = keys.map(k => {
        const raw = localStorage.getItem(k);
        const truncated = raw.length > 200 ? raw.slice(0, 200) + '…' : raw;
        return `<div class="ls-row">
          <div class="ls-key">${k}</div>
          <div class="ls-val">${truncated}</div>
          <div class="ls-actions">
            <button class="dev-btn danger" onclick="deleteKey('${k}')"><i class="fas fa-trash"></i></button>
          </div>
        </div>`;
      }).join('');
    }

    function deleteKey(key) {
      if (!confirm(`Deletar chave "${key}"?`)) return;
      localStorage.removeItem(key);
      renderStorage();
    }

    function exportStorage() {
      const data = {};
      Object.keys(localStorage).forEach(k => { data[k] = localStorage.getItem(k); });
      const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = `pc-localstorage-${new Date().toISOString().slice(0,10)}.json`;
      a.click();
    }

    // Init
    renderOverview();
  </script>
</body>
</html>
```
